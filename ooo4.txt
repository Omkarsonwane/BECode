```cpp
#include <iostream>
#include <omp.h>
using namespace std;

int main() {

    int A[2][2] = {{1,2},{3,4}};
    int B[2][2] = {{5,6},{7,8}};
    int C[2][2];

    #pragma omp parallel for collapse(2)
    for(int i=0;i<2;i++) {
        for(int j=0;j<2;j++) {

            C[i][j] = 0;

            for(int k=0;k<2;k++) {
                C[i][j] += A[i][k] * B[k][j];
            }
        }
    }

    cout << "Matrix Multiplication:" << endl;

    for(int i=0;i<2;i++) {
        for(int j=0;j<2;j++) {
            cout << C[i][j] << " ";
        }
        cout << endl;
    }

    return 0;
}
```

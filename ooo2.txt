#include <iostream>
#include <omp.h>
using namespace std;

void bubbleSort(int arr[], int n) {
    for(int i=0;i<n-1;i++) {

        #pragma omp parallel for
        for(int j=0;j<n-i-1;j++) {
            if(arr[j] > arr[j+1]) {
                swap(arr[j], arr[j+1]);
            }
        }
    }
}

void merge(int arr[], int l, int m, int r) {
    int n1 = m-l+1;
    int n2 = r-m;

    int L[n1], R[n2];

    for(int i=0;i<n1;i++)
        L[i] = arr[l+i];

    for(int j=0;j<n2;j++)
        R[j] = arr[m+1+j];

    int i=0,j=0,k=l;

    while(i<n1 && j<n2) {
        if(L[i] <= R[j])
            arr[k++] = L[i++];
        else
            arr[k++] = R[j++];
    }

    while(i<n1)
        arr[k++] = L[i++];

    while(j<n2)
        arr[k++] = R[j++];
}

void mergeSort(int arr[], int l, int r) {
    if(l<r) {
        int m=(l+r)/2;

        #pragma omp parallel sections
        {
            #pragma omp section
            mergeSort(arr,l,m);

            #pragma omp section
            mergeSort(arr,m+1,r);
        }

        merge(arr,l,m,r);
    }
}

int main() {

    int arr1[] = {64,34,25,12,22,11,90};
    int n1 = 7;

    bubbleSort(arr1,n1);

    cout << "Bubble Sort: ";
    for(int i=0;i<n1;i++)
        cout << arr1[i] << " ";

    cout << endl;

    int arr2[] = {38,27,43,3,9,82,10};
    int n2 = 7;

    mergeSort(arr2,0,n2-1);

    cout << "Merge Sort: ";
    for(int i=0;i<n2;i++)
        cout << arr2[i] << " ";

    return 0;
}
